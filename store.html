import turtle
import time
import builtins
import random

# ---------- НАСТРОЙКИ ОКНА ----------
turtle.setup(800, 600)
turtle.title("🏪 Магазин на Turtle")
turtle.tracer(0)
turtle.hideturtle()

# ---------- ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ ----------
balance = 1000
count = 5
price = 100
sell = 150
shop = "Магазин"
product = "Товар"
_message = ""
_message_time = 0

# ---------- ФУНКЦИИ ОТРИСОВКИ ----------
def update_display():
    """Перерисовывает интерфейс с текущими значениями переменных"""
    turtle.clear()
    _draw_background()
    _draw_text(0, 210, f"🏪 {shop} 🏪", 28, "#4a4a4a", "center")
    _draw_text(-280, 150, f"💰 Баланс: {balance} руб.", 18, "#2e7d32")
    _draw_text(50, 150, f"📦 {product}: {count} шт.", 18, "#8d6e63")
    _draw_text(-280, 100, f"Цена покупки: {price} руб.", 16, "#666666")
    _draw_text(50, 100, f"Цена продажи: {sell} руб.", 16, "#666666")
    _draw_text(0, 20, "Твоя программа управляет магазином!", 16, "#333333", "center")

    y_pos = -150
    if balance > 10000:
        _draw_text(0, y_pos, "⭐ ВЫ БОГАТЫ! ⭐", 22, "#ffd700", "center")
        y_pos -= 30
    if count > 10:
        _draw_text(0, y_pos, "🎁 СКИДКА 10% на след. покупку!", 18, "#9c27b0", "center")
        y_pos -= 30

    if _message and time.time() - _message_time < 3:
        _draw_text(0, -200, _message, 16, "#ff5722", "center")

    turtle.update()

def _draw_background():
    turtle.penup()
    turtle.goto(-350, 250)
    turtle.pendown()
    turtle.fillcolor("#e6f3ff")
    turtle.begin_fill()
    for _ in range(2):
        turtle.forward(700)
        turtle.right(90)
        turtle.forward(500)
        turtle.right(90)
    turtle.end_fill()
    turtle.penup()

def _draw_text(x, y, text, size=14, color="black", align="left"):
    turtle.goto(x, y)
    turtle.color(color)
    turtle.write(text, align=align, font=("Arial", size, "bold"))

# ---------- ПЕРЕОПРЕДЕЛЁННЫЙ PRINT ----------
original_print = builtins.print
def print(*args, **kwargs):
    global _message, _message_time
    _message = ' '.join(str(arg) for arg in args)
    _message_time = time.time()
    update_display()
builtins.print = print

# ---------- ФУНКЦИЯ ВВОДА ----------
def input(prompt="Введите значение:", title="Ввод"):
    return turtle.textinput(title, prompt) or ""
